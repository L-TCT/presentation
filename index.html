Le repository


							class GameRepository extends ServiceEntityRepository
							{
								public function birthJPg()
									{
									$query = $this->getEntityManager()->createQuery('
									SELECT g FROM App\Entity\Game g WHERE 
									DAY(NOW())=DAY(g.releaseJP) 
									AND MONTH(NOW())=MONTH(g.releaseJP) 
									order by g.name ASC');
									return $query->getResult();
									}
								}

							import Acc from "../../../api/Acc";

								export const getAccs = ({ commit }) => {
									Acc
									.allAccs()
									.then(response => { commit('SET_ACCS', response.data['hydra:member']);})
								}

								export const getAcc = ({ commit }, accId) => {
									commit('SET_LOADINGSTATUS', true)
									Acc
									.showAcc(accId)
									.then(response => { commit('SET_ACC', response.data), commit('SET_LOADINGSTATUS', false);})
								}

								export const getNbrAccs = ({ commit }) => {
									Acc
									.allAccs()
									.then(response => { commit('SET_NBRACCS', response.data['hydra:totalItems']);})
								}

								{
									return state.birthdayJPg.filter(birthdayJPg => 
									birthdayJPg.releaseJP !== null)
									.map(event => {
										const events= {};
										events.title = 'JP' +' '+ event.name +' '+ event.platform.name;
										events.rrule = {
											'freq': 'yearly',
											'dtstart': event.releaseJP,
											'until': '2100-01-01'
										}
										events.url = 'https://collection.gouaig.fr/single/Jeux/'+ event.id 
										events.allDay = true;
										events.backgroundColor = 'white';
										events.borderColor = 'red';
										return events;
									})
								}

							class AccessoryUnitTest extends TestCase
								{
									public function testIsTrue()
									{
										$accessory = new Accessory();
										$brand = new Brand();
										$serie = new Serie;

										$accessory->setName('name')
												->setFormat('Loose')
												->setSlug('slug')
												->setPhoto('photo')
												->setBlog('blog')
												->setSerie($serie)
												->setBrand($brand);
										
										$this->assertTrue($accessory->getName() === 'name');
										$this->assertTrue($accessory->getFormat() === 'Loose');
										$this->assertTrue($accessory->getSlug() === 'slug');
										$this->assertTrue($accessory->getPhoto() === 'photo');
										$this->assertTrue($accessory->getBlog() === 'blog');
										$this->assertTrue($accessory->getSerie() === $serie);
										$this->assertTrue($accessory->getBrand() === $brand);
									}

									public function testIsFalse()
									{
										$accessory = new Accessory();
										$brand = new Brand();
										$serie = new Serie;

										$accessory->setName('name')
												->setFormat('format')
												->setSlug('slug')
												->setPhoto('photo')
												->setBlog('blog')
												->setSerie($serie)
												->setBrand($brand);
										
										$this->assertFalse($accessory->getName() === 'false');
										$this->assertFalse($accessory->getFormat() === 'format');
										$this->assertFalse($accessory->getSlug() === 'false');
										$this->assertFalse($accessory->getPhoto() === 'false');
										$this->assertFalse($accessory->getBlog() === 'false');
										$this->assertFalse($accessory->getSerie() === new Serie());
										$this->assertFalse($accessory->getBrand() === new Brand());
									}

									public function testIsEmpty()
									{
										$accessory = new Accessory();

										$this->assertEmpty($accessory->getName());
										$this->assertEmpty($accessory->getFormat());
										$this->assertEmpty($accessory->getSlug());
										$this->assertEmpty($accessory->getPhoto());
										$this->assertEmpty($accessory->getBlog());
										$this->assertEmpty($accessory->getSerie());
										$this->assertEmpty($accessory->getBrand());

									}
								}
